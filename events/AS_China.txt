country_event = {
	id = as_china.1
	title = as_china.1.t
	desc = as_china.1.d
	picture = GFX_report_event_chinese_japanese_handshake

	fire_only_once = yes
	
	trigger = {	
		tag = CHI
		CHI = { exists = yes }
		PRC = { exists = yes } 
		date > 1936.12.10	# histo 12.24			
		NOT = { has_country_flag = chi_xian_incident }
	}
	
	mean_time_to_happen = {
		days = 1
	}	

	immediate = {
			every_state = {
					limit = {	
						is_controlled_by = PRC
						is_owned_by = CHI
						is_core_of = PRC
					}
				PRC = {
					transfer_state = PREV
				}
			}
			every_state = {
					limit = {	
						is_controlled_by = CHI
						is_owned_by = PRC
						is_core_of = CHI
					}
				CHI = {
					transfer_state = PREV
				}
			}			
	}
		
	option = {
		name = as_china.1.a			#Accept the innevitable!
		set_country_flag = chi_xian_incident
		add_national_unity = 0.20
		add_political_power = 300
		random_other_country = {
				limit = {
					tag = PRC
					has_war_with = CHI
				}
				CHI = { white_peace = PRC }				
		}			
		effect_tooltip = {
			PRC = { remove_opinion_modifier = { target = CHI modifier = hostile_status } }
			CHI = { remove_opinion_modifier = { target = PRC modifier = hostile_status } }
		}		
		CHI = {												### AS
				if = {
					limit = {
						CHI = { is_ai = yes }
						NOT = {
								has_idea = war_economy
								has_idea = tot_economic_mobilisation						
						}
					}
						add_ideas = war_economy
				}			
		}
		CHI = {												### AS
				if = {
					limit = {
						CHI = { is_ai = yes }
						NOT = {
								has_idea = service_by_requirement
								has_idea = all_adults_serve
								has_idea = scraping_the_barrel											
						}
					}
						add_ideas = service_by_requirement
				}			
		}		
		diplomatic_relation = {
				country = PRC
				relation = non_aggression_pact
				active = yes
		}
		every_owned_state = {
			if = {
				limit = { 
					is_border_conflict = yes
				}
				set_border_war = no
			}
		}
		PRC = { country_event = { id = as_china.2 } } 
	}
}
