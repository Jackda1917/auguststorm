# Limited War in China
country_event = {
	id = as_japan.3
	title = as_japan.3.t
	desc = as_japan.3.d
	picture = GFX_report_event_japanese_transport_soldiers
	
	trigger = {
		tag = JAP
		has_war_with = CHI
		CHI = { surrender_progress > 0.55 }                 
		NOT = { has_global_flag = chi_limited_war_flag }
	}

	mean_time_to_happen = { days = 1 }				

	fire_only_once = yes

	option = {
		name = as_japan.3.a
		ai_chance = {
			factor = 10
		}
		set_global_flag = chi_limited_war_flag		
		custom_effect_tooltip = as_japan.3_TT	
				#add_ideas = warlords_corruption			
	}
	#Let's Grab it all !
	option = {
		name = as_japan.3.b
		ai_chance = {
			factor = 0
		}
		effect_tooltip = {
			add_political_power = -200
			PRC = { add_ideas = chinese_partisans }
			show_ideas_tooltip = chinese_partisans
			PRC = {
				add_tech_bonus = {   # was none			
					bonus = 0.50
					uses = 3
					category = land_doctrine
				}			
			}
			CHI = {
				add_tech_bonus = {   # was none			
					bonus = 0.50
					uses = 2
					category = land_doctrine
				}			
			}			
		}				
	}
}

# Japan puppet China
country_event = {
	id = as_japan.4
	title = as_japan.4.t
	desc = as_japan.4.d
	picture = GFX_report_event_japanese_transport_soldiers
	
	is_triggered_only = yes  # on_action
	
	immediate = {
		annex_country = { target = PRC transfer_troops = no }
		annex_country = { target = CHI transfer_troops = no }
	}
	
		
	option = {
		name = as_japan.4.a
		clr_global_flag = chi_limited_war_flag
		set_state_controller = 594 
		set_state_controller = 609
		if = {
			limit = { 
				MEN = { exists = no } 
			}
			JAP = { release_puppet = MEN }			
		}
		if = {
			limit = { 
				MAN = { exists = no } 
			}
			JAP = { release_puppet = MAN }			
		}
		if = {
			limit = { 
				CHI = { exists = no } 
			}
			JAP = { release_puppet = CHI }			
		}		
			
		MEN = {		
			transfer_state = 612			
			transfer_state = 611			
			transfer_state = 615
			transfer_state = 621			
			transfer_state = 622
			transfer_state = 283
			transfer_state = 615			
			transfer_state = 616			
		}
		every_state = {
			limit = {
				is_claimed_by = JAP	
				is_core_of = PRC       # Because CHI is now considered as NJG
				NOT = { is_core_of = TIB }
				NOT = { is_core_of = SIK }
			}

			JAP = {
				transfer_state = PREV
				transfer_state = 609 #East Hebei
			}
		}
			if = {
				limit = { has_dlc = "Death or Dishonor" }
					set_autonomy = { 
						target = CHI 
						autonomy_state = autonomy_puppet
						freedom_level = 0.5		
					}

			}
			if = {
				limit = { has_dlc = "Death or Dishonor" }
					set_autonomy = { 
						target = MAN 
						autonomy_state = autonomy_integrated_puppet
						freedom_level = 0.3	
					}

			}
			if = {
				limit = { has_dlc = "Death or Dishonor" }
					set_autonomy = { 
						target = MEN 
						autonomy_state = autonomy_integrated_puppet
						freedom_level = 0.2	
					}

			}		
	}	
}
