	title = as_italy.2.t
	desc = as_italy.2.d
	picture = GFX_report_event_generic_rally2

	fire_only_once = yes

	trigger = {
		tag = ITA		
		enemies_strength_ratio > 1
		has_war_with = USA
		OR = {
			has_national_unity < 0.50
			casualties > 2500000
		}
		OR = {
			any_country {
				is_in_faction_with = USA
				controls_state = 2			
			}
			any_owned_state = {
				is_core_of = ITA
				is_on_continent = europe
				is_controlled_by = ENG
			}
			any_owned_state = {
				is_core_of = ITA
				is_on_continent = europe
				is_controlled_by = CAN
			}			
			any_owned_state = {
				is_core_of = ITA
				is_on_continent = europe				
				is_controlled_by = USA
			}
		}
	}
		
	mean_time_to_happen = {
		days = 5
	}	
	
	immediate = {
		kill_country_leader = yes
	}
	
	option = {
		name = as_italy.2.a
		ai_chance = { factor = 100 }
		news_event = { id = as_news.2 hours = 3 }
		create_country_leader = {
			name = "Pietro Badoglio"
			desc = "POLITICS_PIETRO_BADOGLIO_DESC"
			picture = "Portrait_Italy_Pietro_Badoglio.dds"
			expire = "1965.1.1"
			ideology = conservatism
			traits = {
			}
		}
		set_politics = {
			ruling_party = democratic 
			elections_allowed = yes
		}
		add_popularity = {
			ideology = democratic 
			popularity = 0.35
		}
		add_popularity = {  
			ideology = communism
			popularity = 0.15
		}
		add_popularity = {
			ideology = fascism
			popularity = -0.10
		}			
		if = {
			limit = {
				ENG = { exists = yes }
				ENG = { is_faction_leader = yes }
			}			
			ENG = { country_event = { id = as_italy.3 days = 1 } }
		}
		if = {
			limit = {
				USA = { exists = yes }
				USA = { is_faction_leader = yes }
			}			
			USA = { country_event = { id = as_italy.3 days = 1 } }
		}		
	}
	option = {
		name = as_italy.2.b
		ai_chance = { factor = 0 }
		news_event = { id = as_news.2 hours = 3 }
		start_civil_war = {
			ideology = communism
			size = 0.30
		}
		random_other_country = {
			limit = {
				original_tag = ITA
				has_government = communism
			}
			set_cosmetic_tag = ITA_SOV
			set_equipment_fraction = 0.3
			set_national_unity = 0.9
			create_country_leader = {
				name = "Palmiro Togliatti"
				desc = "POLITICS_PALMIRO_TOGLIATTI_DESC"
				picture = "GFX_Portrait_Italy_Generic_1"
				expire = "1964.08.21"
				ideology = stalinism
				traits = {
					stalins_puppet
				}
			}			
		}		
	}	
}

country_event = {
	id = as_italy.4
	title = as_italy.4.t
	desc = as_italy.4.d
	picture = GFX_report_event_generic_rally2

	fire_only_once = yes
								
	is_triggered_only = yes							# On_actions trigger

	immediate = {
		ENG = { add_to_faction = ITA }		
	}
	
	option = {
		name = as_italy.4.a
		ai_chance = { factor = 100 }
		set_national_unity = 0.90
		add_popularity = {
			ideology = fascism
			popularity = -1
		}		
		every_owned_state = {
			limit = { 
				NOT = { is_core_of = ITA }
			}
			ENG = { transfer_state = PREV }
		}
		every_state = {
			limit = { 
				NOT = { is_core_of = ITA }
				is_controlled_by = ITA
				NOT = { state = 735 }
				NOT = { state = 1 }
			}
			ENG = { transfer_state = PREV }
		}
		if = {
			limit = {
				1 = { 
					is_owned_by = FRA
				}
			}
			FRA = { transfer_state = 1 }
		}		
		if = {
			limit = {
				735 = { 
					is_owned_by = FRA
					NOT = { any_neighbor_state = { is_controlled_by = GER } }
				}
			}
			FRA = { transfer_state = 735 }
			else = { GER = { transfer_state = 735 } }
		}
		country_event = { id = as_italy.9 }		
	}
}
