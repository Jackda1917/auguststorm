country_event = {
	id = as_italy.4
	title = as_italy.4.t
	desc = as_italy.4.d
	picture = GFX_report_event_generic_rally2

	fire_only_once = yes

	is_triggered_only = yes
				
	option = {
		name = as_italy.4.a
		ai_chance = { factor = 100 }
		GER = { remove_from_faction = ITA }
		ENG = { add_to_faction = ITA }
		if = { 
			limit = { has_war_with = ENG } 	
			white_peace = ENG
		}
		white_peace = USA
		if = { 
			limit = { has_war_with = SOV } 	
			white_peace = SOV
		}		
		add_to_war = { targeted_alliance = ENG enemy = GER }		
		add_popularity = {
			ideology = fascism
			popularity = -1
		}
		add_popularity = {
			ideology = democratic 
			popularity = 0.60
		}
		add_popularity = {  
			ideology = communism
			popularity = 0.25
		}	
		add_popularity = {  
			ideology = neutrality
			popularity = 0.15
		}		
		every_owned_state = {
			limit = { 
				NOT = { is_core_of = ITA }
				is_owned_by = ITA
			}
			ENG = { transfer_state = PREV }
		}
		every_owned_state = {
			limit = { 
				NOT = { is_core_of = ITA }
				is_controlled_by = ITA
				NOT = { state = 735 }
				NOT = { state = 1 }
			}
			ENG = { transfer_state = PREV }
		}
		if = {
			limit = {
				1 = { 
					is_owned_by = FRA
				}
			}
			FRA = { transfer_state = 1 }
		}		
		if = {
			limit = {
				735 = { 
					is_owned_by = FRA
					NOT = { any_neighbor_state = { is_controlled_by = GER } }
				}
			}
			FRA = { transfer_state = 735 }
		}
		if = {
			limit = {
				735 = { 
					any_neighbor_state = { is_controlled_by = GER } 
				}
			}
			GER = { transfer_state = 735 }
		}		
		every_other_country  = {	
			every_owned_state = { 
				limit = { 
					is_claimed_by = ITA				
				}
				ITA = { remove_state_claim = PREV }
			}
		}
		start_civil_war = {
				ideology = fascism
				size = 0.30
				capital = 159
				states = { 159 161 158 162 157 2 }
		}
		news_event = { id = as_news.3 hours = 3 }
		random_other_country = {
			limit = {
				original_tag = ITA
				has_government = fascism
			}
			set_equipment_fraction = 0.3
			set_national_unity = 0.8
			set_cosmetic_tag = ITA_RSI			
			create_country_leader = {
				name = "Benito Mussolini"
				desc = "POLITICS_BENITO_MUSSOLINI_DESC"

				ideology = fascism_ideology
				traits = {
					militarily_incompetent # was none
