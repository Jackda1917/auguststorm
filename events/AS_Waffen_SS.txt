# 1. SS-Panzer-Division 'Leibstandarte SS Adolf Hitler'
country_event = {
	id = as_waffen_ss.1
	title = as_waffen_ss.1.title
	desc = as_waffen_ss.1.desc
	picture = GFX_report_event_german_troops
	
	fire_only_once = yes
	
	trigger = {							# Was more or less a brigade before that
		tag = GER
		has_government = fascism
		16 = {
			OR = {
				is_controlled_by = GER
				any_other_country = {
					controls_state = 16 # Paris
					is_in_faction_with = GER
					GER = { is_faction_leader = yes }
				}
			}
		}		
		10 = {
			OR = {
				is_controlled_by = GER
				any_other_country = {
					controls_state = 10 # Warzawa
					is_in_faction_with = GER
					GER = { is_faction_leader = yes }
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = as_waffen_ss.1.option_a
		ai_chance = { factor = 100 }
		as_add_waffen_ss_effect = yes
		as_new_waffen_ss_division_effect = yes
		hidden_effect = { load_oob = GER_Waffen_SS_1 }
		set_country_flag = as_germany_ss_panzer_division_40
		army_experience = -100
	}
	
	option = {
		name = as_waffen_ss.1.option_b
		ai_chance = { factor = 0 }
		add_political_power = -100
		custom_effect_tooltip = GER_waffen_ss_will_never_be_formed
	}
}
