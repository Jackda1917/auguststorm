end_all_production = {
	enable = {
		OR = {
			# Stop all production of units if the country has maxed it's cap
			as_hyper_power_production_maxed = yes 	# 500+ IC
			as_super_power_production_maxed = yes 	# 400-500 IC
			as_great_power_production_maxed = yes 	# 300-400 IC
			as_major_power_production_maxed = yes 	# 200-400 IC
			as_middle_power_production_maxed = yes 	# 100-200 IC
			as_minor_power_production_maxed = yes 	# 50-100 IC
			as_micro_power_production_maxed = yes 	# 25-50 IC
			as_nano_power_production_maxed = yes 	# <25 IC
		}
	}
	abort = {
		# If no cap has been breached, resume production
		OR = {	
			AND = {	
				as_hyper_power_production_maxed = no 	# 500+ IC
				as_super_power_production_maxed = no 	# 400-500 IC
				as_great_power_production_maxed = no 	# 300-400 IC
				as_major_power_production_maxed = no 	# 200-400 IC
				as_middle_power_production_maxed = no 	# 100-200 IC
				as_minor_power_production_maxed = no 	# 50-100 IC
				as_micro_power_production_maxed = no 	# 25-50 IC
				as_nano_power_production_maxed = no 	# <25 IC
			}
			has_idea = war_economy
			has_idea = tot_economic_mobilisation
		}
	}

	ai_strategy = {
		type = role_ratio
		id = infantry
		value = -100
	}

	ai_strategy = {
		type = role_ratio
		id = paratroopers
		value = -100
	}

	ai_strategy = {
		type = role_ratio
		id = mountaineers
		value = -100
	}

		type = role_ratio
